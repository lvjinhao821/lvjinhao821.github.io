案例概况

在输入区输入数字，然后按下代表不同运算方法的按钮，然后进行计算，将结果显示出来。

//获取元素
var calculator = document.querySelector('#calculator');
var formerInput = calculator.querySelector('.formerInput');
var laterInput = calculator.querySelector('.laterInput');
var sign = calculator.querySelector('.sign');
var resultOutput = calculator.querySelector('.resultOutput');

//加法
function addHandler(){
    sign.innerHTML = '+';
    resultOutput.innerHTML = +formerInput.value +     +laterInput.value;
}
// 减法
function subtractHandler(){
    sign.innerHTML = '-';
    resultOutput.innerHTML = formerInput.value - laterInput.value;
}
// 乘法
function multiplyHandler(){
    sign.innerHTML = '*';
    resultOutput.innerHTML = formerInput.value * laterInput.value;
}
// 除法
function divideHandler(){
    sign.innerHTML = '/';
    resultOutput.innerHTML = formerInput.value / laterInput.value;
}

优化1：结构和行为分离

不要将结构、样式和行为糅杂在一起，而是应该将他们彼此分离开，HTML只表示结构、样式交给CSS处理、行为交给JS处理。

事件绑定应该属于网页的行为，因此事件绑定应该放在js文件里。

//结构和行为分离
var btns = calculator.querySelectorAll('.btn')
//绑定事件
btns[0].onlick = addHandler;
btns[1].onlick = subtractHandler;
btns[2].onlick = multiplyHandler;
btns[3].onlick = divideHandler;

优化2：使用循环

明显有规律性且类似于数组，可以利用循环来完成.

给按钮，添加了title属性

<p>
    <input type="button" value="+" class="btn" title="add"/>
    <input type="button" value="-" class="btn" title="subtract"/>
    <input type="button" value="*" class="btn" title="multiply"/>
    <input type="button" value="/" class="btn" title="divide"/>
</p>

根据title属性值的不同，判断点击后运行什么函数

// 使用循环
for(var i = 0; i < btns.length; i++) {
    btns[i].onclick = function() {
        switch(this.title) {
            case 'add':
                addHandler();
                break;
            case 'subtract':
                subtractHandler();
                break;
            case 'multiply':
                multiplyHandler();
                break;
            case 'divide':
                divideHandler();
                break;
        }
    }
}

优化3：提取函数

如果函数暴露了过多的细节，就可用函数封装起来

封装for循环

each(btns,function(index, elem){
    elem.onclick = function() {
        switch(this.title) {
            case 'add':

                subtractHandler();
                break;
            case 'multiply':
                multiplyHandler();
                break;
            case 'divide':
                divideHandler();
                break;
        }
    }
})
//遍历
function each(array, fn) {
    for(var i = 0; i< array.length; i++){
        fn(i, array[i])
    }
}
更新符号

一个函数尽量只做一件事，否则会变得很复杂，其功能和职责也是不明确的，因此+-*/都分别写一个函数。

//更新符号
function updateSign(symbol){
    sign.innerHTML = symbol
}
//加法 
function add(num1, num2){
    return +num1 + +num2  //加个+号，为了把字符串变为数字
}
//减法 
function subtract(num1, num2){
    return num1 - num2
}
//乘法 
function multiply(num1, num2){
    return num1 * num2
}
//除法 
function divide(num1, num2){
    return num1 / num2
}
//输出结果
function outputResult(result){
    resultOutput.innerHTML = result;
}


//加法
function addHandler(){
    updateSign('+');
    outputResult(add(formerInput.value, laterInput.value));
}
// 减法
function subtractHandler(){
    updateSign('-');
    outputResult(subtract(formerInput.value, laterInput.value));
}
// 乘法
function multiplyHandler(){
    updateSign('*');
    outputResult(multiply(formerInput.value, laterInput.value));
}
// 除法
function divideHandler(){
    updateSign('/');
    outputResult(divide(formerInput.value, laterInput.value));
}

下图函数其实做了两件事，每次点击时更新符号，以及计算出结果。这两件事都可以放在遍历每个按钮的时候。

在遍历时加入符号更新和计算结果更新

优化4：管理代码

如图设置了很多全局变量，但他们却是有联系的，可以为这样的代码找到组织。通过创建一个对象，然后与之相关的变量，变为其属性。从属关系也会更加明确，也便于代码管理。

//获取元素
var wrapElem = document.querySelector('#calculator');
var calculatorElem = {
    formerInput: calculator.querySelector('.formerInput'),
    laterInput: calculator.querySelector('.laterInput'),
    sign: calculator.querySelector('.sign'),
    resultOutput: calculator.querySelector('.resultOutput'),
    btns: calculator.querySelectorAll('.btn')
};

改完后，记得把之前代码中用到的全局变量，变为calculatorElem.全局变量。

前面说道没把+-*/函数放到同一个函数里，而是选择了分开写，但分开的小函数越多却越难管理，因此最好给他们找一个组织，既然他们都表示运算，就把他们放在一个对象里。

var operation = {
    add: function(num1, num2){
            return +num1 + +num2
        },
    subtract: function(num1, num2){
            return num1 - num2
        },
    multiply: function(num1, num2){
            return num1 * num2
        },
    divide: function(num1, num2){
            return num1 / num2
        }
};

改完后，记得吧之前代码中用到的函数名，变为operation.变量名。

优化5：开放和封闭原则

开放封闭原则（OCP，Open Closed Principle）是所有面向对象原则的核心。指的是对扩展开放，对修改封闭。

  • 对扩展开放，意味着有新的需求或变化时，可以对现有代码进行扩展，以适应新情况。
  • 对修改封闭，意味着类一旦设计完成，就可以独立完成其工作，而不要对类进行任何修改。

对添加功能开放，对修改代码封闭。 通常就是提供一个接口，要增加功能就通过这个接口，绝不能修改原有的代码。

本例中，大量使用switch是很糟糕的，往往使得代码不容易修改，如果需求变了，需要添加一个功能，就只能在原有代码内部去修改。

这里switch的作用仅仅是根据点击的符号，来判断使用哪个计算方法。可以新增一个函数来代替它。

each(calculatorElem.btns,function(index, elem){
    elem.onclick = function() {
        updateSign(this.value)
        outputResult(operate(this.title, calculatorElem.formerInput.value, calculatorElem.laterInput.value));
    }
})

//运算
function operate(name, num1, num2){
    //手动抛出错误，就是为了方便我们遇到问题时，能更准确定位到错误的问题↓
    if(!operation[name]) throw new Error('不存在名为' + name + '的运算方法');
    //此时的name肯定就是存在于operation对象里了
    return operation[name](num1, num2);
}

如果需求有更改，需要添加一个求模(%)方法的按钮，在HTML里增加这个按钮，其功能只需要在JS中operation运算符对象里面添加一个求模方法，为避免直接在原码处添加，就需要提供一个接口在外进行添加。

优化6：模块化

我们这里写的是计算器的程序，计算器的程序可以当成一个模块。如果把这个模板，在JS代码中像现在一样写，很明显它暴露出了太多全局变量，而且其他人并不关心里面的变量，这些变量暴露出来反而污染了全局作用域。大家只想知道这个模块是怎么调用的，或者直接调用了就好。这时候就可以利用匿名函数的自执行，这样既能够执行，又能够放到局部作用域中使用，与外面的代码互不影响。

(function(){
    以上全部函数
})();

本例中operate函数和operation对象，两个在一起和核心作用就是，接收参数、进行相应计算、然后返回结果。这个功能是一个比较独立的部分，因此可以做成一个模块，提供一个接口，大家通过这个接口来使用就行。

var operate = (function(){
        var operation = {
            add: function(num1, num2){
                    return +num1 + +num2;
                },
            subtract: function(num1, num2){
                    return num1 - num2;
                },
            multiply: function(num1, num2 ){
                    return num1 * num2;
                },
            divide: function(num1, num2){
                    return num1 / num2;
                },
            addOperation: function(name, fn){
                if(!operation[name]){
                    operation[name] = fn;  //如果新增的方法，对象里没有，才可以添加此方法
                }
                return operation;       //让这个函数可以链式调用
            }
        };

        //运算
        function operate(name, num1, num2){
            //手动抛出错误，就是为了方便我们遇到问题时，能更准确定位到错误的问题↓
            if(!operation[name]) throw new Error('不存在名为' + name + '的运算方法');
            //此时的name肯定就是存在于operation对象里了
            return operation[name](num1, num2);
        }  

        operate.addOperation = operation.addOperation;
        return operate;
})()
//通过对外接口，进行添加求模(%)函数
operate.addOperation('mod',function(num1, num2){
    return num1 % num2;
});

现在这个模块调用的时候，需要传3个参数（name、num1、num2），可是为什么只能传入num1 和 num2两个参数呢！万一需要传入的参数数量比较多，程序就没法接收完参数了。

此时最好的办法是不明确指定要传入的参数，可利用arguments代表实参且它是类数组的特点。考虑如何将一个类数组，作为一个参数，传入进来。

修改和建议

  • 对输入进来的值，要进行验证。
    • 只要用表单都需要验证，验证其实也是比较独立的模块。可以尝试用模块的写法，将验证写为独立的模块，然后提供一个借口供计算器使用就好。
  • 整数的运算没有问题，可JS对小数的计算存在问题，因此对于小数的运算要进行处理。
    • 可以把小数乘以多少个0变为整数，然后进行运算，最后在除以这么多0就好。
  • 现在运算要靠鼠标点击对应的按钮才行，如果能监控键盘，如果按下了对应的键就可进行运算就更好。
  • 本例的绑定事件，用的是循环遍历，然后分别为其绑定事件。这种方式如果按钮很多，效率就会很低。因此如果按钮很多就可以用事件代理的方法，就只用绑定一次，不管数量有多少，就算通过js添加删除按钮，都只用绑定一次，效率会很高。

                },

                    return num1 - num2;
